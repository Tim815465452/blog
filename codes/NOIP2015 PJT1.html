To solve this problem, all we need to do is to simulate the scene because the data range is so small that we dont't any algorithms to simplify it.

Code

Pascal

var
    k,i,j:integer;
    ans:longint;
begin
    read(k);
    ans:=0;
    i:=1;
    while k>0 do
        begin
            for j:=1 to i do
                if k>0 then
                    begin
                        ans:=ans+i;
                        dec(k);
                    end;
            inc(i);
        end;
    write(ans);
end.

C++

#include<cstdio>
int main()
{
    int i=0,k;
    long ans=0;
    scanf("%d",&k);
    while(k>0)
    {
        for(int j=0;j<i;j++)
            if(k>0)
            {
                k--;
                ans+=i;
            }
        i++;
    }
    printf("%d",ans);
    return 0;
}
