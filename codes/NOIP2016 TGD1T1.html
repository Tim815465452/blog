Solution

To solve this problem, we just need to simulate the scene, and decide if Xiao Nan should move clockwise or counterclockwise.

Code

Pascal

const
    MAXN=100000;
type
    toy=record
        job:string;
        face:integer;
    end;
var
    a:array[1..MAXN]of toy;
    n,m,i,p,d,s:longint;
begin
    readln(n,m);
    for i:=1 to n do
        readln(a[i].face,a[i].job);
    p:=1;//Start from the 1st toy
    for i:=1 to m do
        begin
            readln(d,s);
            if a[p].face=d then
                p:=(p+n-s-1) mod n+1//Move clockwise
            else
                p:=(p+s-1) mod n+1;//Move counterclockwise
        end;
    writeln(copy(a[p].job,2,20));
end.

C++

#include<cstdio>
#define MAXN 100000
int face[MAXN];
char job[MAXN][20];
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)
        scanf("%d%s",&face[i],&job[i]);
    int p=0;//Start from the 1st toy
    for(int i=0;i<m;i++)
    {
        int d,s;
        scanf("%d%d",&d,&s);
        if(face[p]==d)
            p+=n-s;//Move clockwise
        else
            p+=s;//Move counterclockwise
        p%=n;
    }
    printf("%s",job[p]);
    return 0;
}
